    (function () {

      var correctPassword = "IMCHECKn4krys";

      var gate = document.getElementById("gate");

      var input = document.getElementById("passInput");

      var button = document.getElementById("gateButton");

      var error = document.getElementById("gateError");

      var music = document.getElementById("bgMusic");

      var playBtn = document.getElementById("playMusicBtn");


      function showPlayButton() {

        if (playBtn) {

          playBtn.style.display = "block";

        }

      }


      function tryUnlock() {

        if (input.value.trim() === correctPassword) {

          gate.style.display = "none";

          content.classList.remove("content-hidden");

          window.scrollTo(0, 0);

          showPlayButton();

        } else {

          error.textContent = "That doesn't look quite right. Try again.";

          input.value = "";

          input.focus();

        }

      }


      if (button) {

        button.addEventListener("click", tryUnlock);

      }


      if (input) {

        input.addEventListener("keydown", function (e) {

          if (e.key === "Enter") {

            tryUnlock();

          }

        });

      }


      if (playBtn && music) {

        playBtn.addEventListener("click", function () {

          if (music.paused) {

            music.play();

            playBtn.textContent = "❚❚ Pause Music";

            playBtn.style.opacity = 0.9;

          } else {

            music.pause();

            playBtn.textContent = "▶ Play Music";
